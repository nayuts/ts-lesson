  id: number;
  title: string;
  status: "todo" | "doing" | "done";
  description?: string;
}

const todoList: Todo[] = [
  {
    id: 1,
    title: "ご飯を炊く",
    status: "done",
  },
  {
    id: 2,
    title: "フライパンを出す",
    status: "todo",
    description: "魚を焼くために必要",
  },
];

const targetUser = todoList.find((todo) => todo.id === 1);
console.log(targetUser);

const incompleteTodos = todoList.filter((todo) => todo.status !== "done");
console.log(incompleteTodos);
